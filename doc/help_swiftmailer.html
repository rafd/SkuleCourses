$connection = new Swift_Connection_SMTP($mailNotificationParams['sender_smtp'], 465, Swift_Connection_SMTP::ENC_SSL));
$connection->setUsername("username");
$connection->setPassword("password);

$mailer = new Swift($connection);
$message = new Swift_Message("A test msg", "Hello world!");

$recipients = new Swift_RecipientList();
$recipients->addTo("test_recipient@test.com");

$mailer->send($message, $recipients, "test_sender@test.com);
$mailer->disconnect();
